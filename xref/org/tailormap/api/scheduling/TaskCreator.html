6   package org.tailormap.api.scheduling;
7   
8   import java.lang.invoke.MethodHandles;
9   import java.util.UUID;
10  import org.quartz.CronScheduleBuilder;
11  import org.quartz.DateBuilder;
12  import org.quartz.JobBuilder;
13  import org.quartz.JobDataMap;
14  import org.quartz.JobDetail;
15  import org.quartz.JobKey;
16  import org.quartz.ObjectAlreadyExistsException;
17  import org.quartz.Scheduler;
18  import org.quartz.SchedulerException;
19  import org.quartz.Trigger;
20  import org.quartz.TriggerBuilder;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.scheduling.quartz.QuartzJobBean;
25  import org.springframework.stereotype.Service;
26  
27  @Service
28  public class TaskCreator {
29    private static final Logger logger =
30        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
31  
32    private final Scheduler scheduler;
33  
34    public TaskCreator(@Autowired Scheduler scheduler) {
35      this.scheduler = scheduler;
36    }
37  
38    /**
39     * Create a job and schedule it with a cron expression.
40     *
41     * @param job the task class to create
42     * @param jobData a map with job data, the {@code type} and {@code description} keys are mandatory
43     * @param cronExpression the cron expression
44     * @return the task name, a UUID
45     * @throws SchedulerException if the job could not be scheduled
46     */
47    public String createTask(
48        Class<? extends QuartzJobBean> job, TMJobDataMap jobData, String cronExpression)
49        throws SchedulerException {
50  
51      // Create a job
52      JobDetail jobDetail =
53          JobBuilder.newJob(job)
54              .withIdentity(new JobKey(UUID.randomUUID().toString(), jobData.get("type").toString()))
55              .withDescription(jobData.getDescription())
56              .usingJobData(new JobDataMap(jobData))
57              .build();
58  
59      // Create a trigger
60      Trigger trigger =
61          TriggerBuilder.newTrigger()
62              .withIdentity(jobDetail.getKey().getName(), jobDetail.getKey().getGroup())
63              .startAt(DateBuilder.futureDate(30, DateBuilder.IntervalUnit.SECOND))
64              .withPriority(jobData.getPriority())
65              .withSchedule(
66                  CronScheduleBuilder.cronSchedule(cronExpression)
67                      .withMisfireHandlingInstructionFireAndProceed())
68              .build();
69  
70      try {
71        scheduler.scheduleJob(jobDetail, trigger);
72      } catch (ObjectAlreadyExistsException ex) {
73        logger.warn(
74            "Job {} with trigger {} has not bean added to scheduler as it already exists.",
75            jobDetail.getKey(),
76            trigger.getKey());
77        return null;
78      }
79  
80      return jobDetail.getKey().getName();
81    }
82  }
