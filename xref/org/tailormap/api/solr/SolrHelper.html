45  /**
46   * Solr utility/wrapper class. This class provides methods to add or update a full-text feature type
47   * index for a layer, find in the index for a layer, and clear the index for a layer. It also
48   * provides a method to close the Solr client as well as automatically closing the client when used
49   * in a try-with-resources.
50   */
51  public class SolrHelper implements AutoCloseable, Constants {
52    private final SolrClient solrClient;
53  
54    private static final Logger logger =
55        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
56  
57    public static final int SOLR_BATCH_SIZE = 1000;
58    // milliseconds
59    private static final int SOLR_TIMEOUT = 7000;
60  
61    /**
62     * Constructor
63     *
64     * @param solrClient the Solr client, this will be closed when this class is closed
65     */
66    public SolrHelper(@NotNull SolrClient solrClient) {
67      this.solrClient = solrClient;
68    }

70    /**
71     * Add or update a feature type index for a layer.
72     *
73     * @param searchIndex the search index config
74     * @param tmFeatureType the feature type
75     * @param featureSourceFactoryHelper the feature source factory helper
76     * @param searchIndexRepository the search index repository, so we can save the {@code
77     *     searchIndex}
78     * @throws IOException if an I/O error occurs
79     * @throws SolrServerException if a Solr error occurs
80     * @return the possibly updated {@code searchIndex} object
81     */
82    @SuppressWarnings("FromTemporalAccessor")
83    public SearchIndex addFeatureTypeIndex(
84        @NotNull SearchIndex searchIndex,
85        @NotNull TMFeatureType tmFeatureType,
86        @NotNull FeatureSourceFactoryHelper featureSourceFactoryHelper,
87        @NotNull SearchIndexRepository searchIndexRepository)
88        throws IOException, SolrServerException {
89  
90      createSchemaIfNotExists();
91  
92      final Instant start = Instant.now();
93  
94      if (null == searchIndex.getSearchFieldsUsed()) {
95        logger.warn(
96            "No search fields configured for search index: {}, bailing out.", searchIndex.getName());
97        return searchIndexRepository.save(
98            searchIndex
99                .setStatus(SearchIndex.Status.ERROR)
100               .setComment("No search fields configured"));
101     }
102 
103     // set fields while filtering out hidden fields
104     List<String> searchFields =
105         searchIndex.getSearchFieldsUsed().stream()
106             .filter(s -> !tmFeatureType.getSettings().getHideAttributes().contains(s))
107             .toList();
108     List<String> displayFields =
109         searchIndex.getSearchDisplayFieldsUsed().stream()
110             .filter(s -> !tmFeatureType.getSettings().getHideAttributes().contains(s))
111             .toList();
112 
113     if (searchFields.isEmpty()) {
114       logger.warn(
115           "No valid search fields configured for featuretype: {}, bailing out.",
116           tmFeatureType.getName());
117       return searchIndexRepository.save(
118           searchIndex
119               .setStatus(SearchIndex.Status.ERROR)
120               .setComment("No search fields configured"));
121     }
122 
123     // add search and display properties to query
124     Set<String> propertyNames = new HashSet<>();
125     // always add primary key and default geometry to geotools query
126     propertyNames.add(tmFeatureType.getPrimaryKeyAttribute());
127     propertyNames.add(tmFeatureType.getDefaultGeometryAttribute());
128     propertyNames.addAll(searchFields);
129 
130     final boolean hasDisplayFields = !displayFields.isEmpty();
131     if (hasDisplayFields) {
132       propertyNames.addAll(displayFields);
133     }
134 
135     clearIndexForLayer(searchIndex.getId());
136 
137     logger.info(
138         "Indexing started for index id: {}, feature type: {}",
139         searchIndex.getId(),
140         tmFeatureType.getName());
141     searchIndex = searchIndexRepository.save(searchIndex.setStatus(SearchIndex.Status.INDEXING));
142 
143     // collect features to index
144     SimpleFeatureSource fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
145     Query q = new Query(fs.getName().toString());
146     // filter out any hidden properties (there should be none though)
147     tmFeatureType.getSettings().getHideAttributes().forEach(propertyNames::remove);
148     q.setPropertyNames(List.copyOf(propertyNames));
149     q.setStartIndex(0);
150     // TODO: make maxFeatures configurable? perhaps for WFS sources?
151     // q.setMaxFeatures(Integer.MAX_VALUE);
152     logger.trace("Indexing query: {}", q);
153     SimpleFeatureCollection simpleFeatureCollection = fs.getFeatures(q);
154     final int total = simpleFeatureCollection.size();
155     List<FeatureIndexingDocument> docsBatch = new ArrayList<>(SOLR_BATCH_SIZE);
156     // TODO this does not currently batch/page the feature source query, this doesn't seem to be an
157     //   issue for now but could be if the feature source is very large or slow e.g. WFS
158     UpdateResponse updateResponse;
159     try (SimpleFeatureIterator iterator = simpleFeatureCollection.features()) {
160       int indexCounter = 0;
161       while (iterator.hasNext()) {
162         indexCounter++;
163         SimpleFeature feature = iterator.next();
164         // note that this will create a unique document
165         FeatureIndexingDocument doc =
166             new FeatureIndexingDocument(feature.getID(), searchIndex.getId());
167         List<String> searchValues = new ArrayList<>();
168         List<String> displayValues = new ArrayList<>();
169         propertyNames.forEach(
170             propertyName -> {
171               Object value = feature.getAttribute(propertyName);
172               if (value != null) {
173                 if (value instanceof Geometry) {
174                   doc.setGeometry(GeometryProcessor.processGeometry(value, true, true, null));
175                 } else {
176                   // when display and/or search fields are configured, add the value to the search
177                   // and/or display field otherwise add the value to the search and display field
178                   if (searchFields.contains(propertyName)) {
179                     searchValues.add(value.toString());
180                   }
181                   if (hasDisplayFields) {
182                     if (displayFields.contains(propertyName)) {
183                       displayValues.add(value.toString());
184                     }
185                   }
186                 }
187               }
188             });
189         doc.setSearchFields(searchValues.toArray(new String[searchFields.size() + 2]));
190         doc.setDisplayFields(displayValues.toArray(new String[0]));
191         docsBatch.add(doc);
192         if (indexCounter % SOLR_BATCH_SIZE == 0) {
193           updateResponse = solrClient.addBeans(docsBatch);
194           logger.info(
195               "Added {} documents of {} to index, result status: {}",
196               indexCounter,
197               total,
198               updateResponse.getStatus());
199           docsBatch.clear();
200         }
201       }
202     } finally {
203       if (fs.getDataStore() != null) fs.getDataStore().dispose();
204     }
205 
206     if (!docsBatch.isEmpty()) {
207       updateResponse = solrClient.addBeans(docsBatch);
208       logger.info("Added last {} documents of {} to index", docsBatch.size(), total);
209       logger.debug("Update response status: {}", updateResponse.getStatus());
210     }
211     final Instant end = Instant.now();
212     Duration processTime = Duration.between(start, end).abs();
213     logger.info(
214         "Indexing finished for index id: {}, featuretype: {} at {} in {}",
215         searchIndex.getId(),
216         tmFeatureType.getName(),
217         end,
218         processTime);
219     updateResponse = this.solrClient.commit();
220     logger.debug("Update response status: {}", updateResponse.getStatus());
221     return searchIndex
222         .setComment(
223             "Indexed %s features in %s.%s seconds, started at %s"
224                 .formatted(total, processTime.getSeconds(), processTime.getNano(), start))
225         .setLastIndexed(end.atOffset(ZoneId.systemDefault().getRules().getOffset(end)))
226         .setStatus(SearchIndex.Status.INDEXED);
227   }

229   /**
230    * Clear the index for a layer.
231    *
232    * @param searchLayerId the layer id
233    * @throws IOException if an I/O error occurs
234    * @throws SolrServerException if a Solr error occurs
235    */
236   public void clearIndexForLayer(@NotNull Long searchLayerId)
237       throws IOException, SolrServerException {
238     QueryResponse response =
239         solrClient.query(
240             new SolrQuery("exists(query(" + SEARCH_LAYER + ":" + searchLayerId + "))"));
241     if (response.getResults().getNumFound() > 0) {
242       logger.info("Clearing index for searchLayer {}", searchLayerId);
243       UpdateResponse updateResponse = solrClient.deleteByQuery(SEARCH_LAYER + ":" + searchLayerId);
244       logger.debug("Update response status: {}", updateResponse.getStatus());
245       updateResponse = solrClient.commit();
246       logger.debug("Update response status: {}", updateResponse.getStatus());
247     } else {
248       logger.info("No index to clear for layer {}", searchLayerId);
249     }
250   }

252   /**
253    * Search in the index for a layer. The given query is augmented to filter on the {@code
254    * solrLayerId}.
255    *
256    * @param searchIndex the search index
257    * @param solrQuery the query, when {@code null} or empty, the query is set to {@code *} (match
258    *     all)
259    * @param start the start index, starting at 0
260    * @param numResultsToReturn the number of results to return
261    * @return the documents
262    * @throws IOException if an I/O error occurs
263    * @throws SolrServerException if a Solr error occurs
264    */
265   public SearchResponse findInIndex(
266       @NotNull SearchIndex searchIndex, String solrQuery, int start, int numResultsToReturn)
267       throws IOException, SolrServerException, SolrException {
268     logger.info("Find in index for {}", searchIndex.getId());
269     if (null == solrQuery || solrQuery.isBlank()) {
270       solrQuery = "*";
271     }
272     // TODO We could escape special/syntax characters, but that also prevents using
273     //      keys like ~ and *
274     // solrQuery = ClientUtils.escapeQueryChars(solrQuery);
275 
276     final SolrQuery query =
277         new SolrQuery(INDEX_SEARCH_FIELD + ":" + solrQuery)
278             .setShowDebugInfo(logger.isDebugEnabled())
279             .setTimeAllowed(SOLR_TIMEOUT)
280             .setIncludeScore(true)
281             .setFields(SEARCH_ID_FIELD, INDEX_DISPLAY_FIELD, INDEX_GEOM_FIELD)
282             .addFilterQuery(SEARCH_LAYER + ":" + searchIndex.getId())
283             .setSort("score", SolrQuery.ORDER.desc)
284             .addSort(SEARCH_ID_FIELD, SolrQuery.ORDER.asc)
285             .setRows(numResultsToReturn)
286             .setStart(start);
287     query.set("q.op", "AND");
288     logger.debug("Solr query: {}", query);
289 
290     final QueryResponse response = solrClient.query(query);
291     logger.debug("response: {}", response);
292 
293     final SolrDocumentList solrDocumentList = response.getResults();
294     logger.debug("Found {} solr documents", solrDocumentList.getNumFound());
295     final SearchResponse searchResponse =
296         new SearchResponse()
297             .total(solrDocumentList.getNumFound())
298             .start(response.getResults().getStart())
299             .maxScore(solrDocumentList.getMaxScore());
300     response
301         .getResults()
302         .forEach(
303             solrDocument -> {
304               List<String> displayValues =
305                   solrDocument.getFieldValues(INDEX_DISPLAY_FIELD).stream()
306                       .map(Object::toString)
307                       .toList();
308               searchResponse.addDocumentsItem(
309                   new SearchDocument()
310                       .fid(solrDocument.getFieldValue(SEARCH_ID_FIELD).toString())
311                       .geometry(solrDocument.getFieldValue(INDEX_GEOM_FIELD).toString())
312                       .displayValues(displayValues));
313             });
314 
315     return searchResponse;
316   }

318   /**
319    * Programmatically create (part of) the schema if it does not exist. Only checks for the
320    * existence of the search layer {@link Constants#SEARCH_LAYER}.
321    *
322    * @throws SolrServerException if a Solr error occurs
323    * @throws IOException if an I/O error occurs
324    */
325   private void createSchemaIfNotExists() throws SolrServerException, IOException {
326     SchemaRequest.Field fieldCheck = new SchemaRequest.Field(SEARCH_LAYER);
327     boolean schemaExists = true;
328     try {
329       SchemaResponse.FieldResponse isField = fieldCheck.process(solrClient);
330       logger.debug("Field type {} exists", isField.getField());
331     } catch (Exception e) {
332       logger.debug(e.getLocalizedMessage());
333       logger.info("Field type {} does not exist, creating it", SEARCH_LAYER);
334       schemaExists = false;
335     }
336 
337     if (schemaExists) {
338       return;
339     }
340 
341     logger.info("Creating Solr field type {}", SEARCH_LAYER);
342     SchemaRequest.AddField schemaRequest =
343         new SchemaRequest.AddField(
344             Map.of(
345                 "name", SEARCH_LAYER,
346                 "type", "string",
347                 "indexed", true,
348                 "stored", true,
349                 "multiValued", false,
350                 "required", true,
351                 "uninvertible", false));
352     schemaRequest.process(solrClient);
353 
354     logger.info("Creating Solr field type {}", INDEX_GEOM_FIELD);
355     // TODO https://b3partners.atlassian.net/browse/HTM-1091
356     //  this should be a spatial field type using ("type", "location_rpt")
357     //  but that requires some more work
358     SchemaRequest.AddField schemaRequestGeom =
359         new SchemaRequest.AddField(
360             Map.of(
361                 "name", INDEX_GEOM_FIELD,
362                 "type", "string",
363                 "indexed", false,
364                 "stored", true,
365                 "multiValued", false));
366     schemaRequestGeom.process(solrClient);
367   }
